\section{Pico de Gallo}
  \begin{multicols}{2}
    \subsection{Ingredients}
      \begin{itemize}
        \item 2 medium tomatoes
        \item \sfrac{1}{3} cup fresh coriander
        \item \sfrac{1}{4} cup red onion
        \item Juice of \sfrac{1}{2} a lime
        \item 1/2 tsp salt
      \end{itemize}
  \vfill\null
  \columnbreak
  \subsection{Method}
    \begin{enumerate}
      \item Chop all solid ingredients finely (especially the onione).
      \item Combine in a bowl with the lime juice and salt.
    \end{enumerate}
  \end{multicols}
